function useInnerHTML() {
  document.getElementById("display").innerHTML =
    "This message uses innerHTML.";
}

function useDocumentWrite() {
  document.write("This message uses document.write().");
}

function useAlert() {
  alert("This is an alert message.");
}

function useConsole() {
  console.log("This message is printed in the console.");
}

<!DOCTYPE html>
<html>
<head>
  <title>User Input</title>
</head>
<body>

  <h2>User Greeting</h2>

  <input type="text" id="username" placeholder="Enter your name">
  <input type="text" id="major" placeholder="Enter your major">

  <button onclick="greetUser()">Greet</button>
  <button onclick="clearFields()">Clear</button>

  <p id="greeting"></p>

  <script src="functions.js"></script>
</body>
</html>
function greetUser() {
  let name = document.getElementById("username").value;
  let major = document.getElementById("major").value;

  document.getElementById("greeting").innerHTML =
    "Hello " + name + "! Your major is " + major + ".";
}

function clearFields() {
  document.getElementById("username").value = "";
  document.getElementById("major").value = "";
  document.getElementById("greeting").innerHTML = "";
}
